There are many nice fonts. You can find a lot of fonts at the site of Google Fonts. Here you can download a fonts. In openprocessing you can upload a fonts in the settings menu, tab Files. In the p5 editor you can go to the menu View - Show Sketch Folder. Here you can save your font files in a separate folder /fonts. Then it is easy to copy them to a new sketch.
Now you have to tell p5 that you like to use this awesome font and p5 needs to load this font in memory, before p5 is able to use it. You have to this in the preload function of p5. 
  var myAwesomeFont;
  function preload{
    //In the p5 editor don't forget to include the pathname of your font folder
    //p5
    myAwesomeFont = loadFont("fonts/PrincessSofia-Regular.ttf");
    //openprocessing
    myAwesomeFont = loadFont("PrincessSofia-Regular.ttf");

  }

var fontNice;
function preload() {
   fontNice = loadFont("PrincessSofia-Regular.ttf");
}
function setup() {
  createCanvas(windowWidth, windowHeight);
  background(0);
  textFont(fontNice);
  textSize(40);
  textAlign(CENTER);

  fill(255);
}

function draw() {
  text("The name of this font is PrincessSofia-Regular.",0,0, 400,400);
}
