Let's start with an example with two conditions.
This blue dot is wandering within the borders of the canvas.
We control the borders.

  var x,y;
  function setup(){
    createCanvas(400, 400);
    //start in the center
    x = width/2;
    y = height/2;
  }
  function draw(){
    background(200);
    stroke(0, 191, 255);
    strokeWeight(80);

    //Here you need two conditions.
    if(x > 0 && x < width){
      x += random(-5,5);
    }
    if(y > 0 && y < height){
      y += random(-5,5);
    }
    point(x,y);

  }

But there is something wrong, the blue dot can get stuck in the corners. Do you know why? Think about it. 

  var x,y;
  function setup(){
    createCanvas(400, 400);
    //start in the center
    x = width/2;
    y = height/2;
  }
  function draw(){
    background(200);
    stroke(0, 191, 255);
    strokeWeight(80);

    //Here you need two conditions.
    if(x > 0 && x < width){
      x += random(-5,5);
    }
    else{
      x = width/2;
    }
    if(y > 0 && y < height){
      y += random(-5,5);
    }
    else{
      y = height/2;
    }
    point(x,y);

  }

The painter

There is allways a solution, maybe you can think of another solution to solve this problem.

You can also control your frames with these conditionals. frameCount is a p5 variable that stores a counter for the frame you see. You can use this variable to create scenes or layers, like this: 


  function setup(){
    createCanvas(400, 400);


  }
  function draw(){

    //layer construction
    if(frameCount > 0 && frameCount < 200){
      stroke(0, 191, 255);
      strokeWeight(80);
      point(random(width), random(height),30,30);
    }
    else if(frameCount >= 200 && frameCount < 400){
      stroke(191, 255, 0);
      strokeWeight(5);
      noFill();
      ellipse(random(width), random(height),60,60);
    }
    else if(frameCount >= 400 && frameCount < 600){
      stroke(0, 191, 255);
      strokeWeight(5);
      noFill();
      rect(random(width), random(height),40,40);
    }
    else if(frameCount >= 600 && frameCount < 800){
      noStroke();
      fill(255, 191, 0);
      rect(random(width), random(height),40,40);
    }
    else{           // without a condition
      noLoop();
    }

  }
