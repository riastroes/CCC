Lesson 16: Images

var x, y;
function setup() {
createCanvas(400,300);
background(100);
x = 0;
y =100;
}

function draw() {
stroke(200,255,255);
fill(255,255,200);
for(var i = 0; i < 4; i++){
x = i * 50;        //i = 0, 1, 2, 3
ellipse(x, y, 30,30);
rect(x-15,y+15,30,60);
}
println(i);
}

open sketch in openprocessing

In the for function you can use this counter (i), but don't change this counter it self. (tip)
