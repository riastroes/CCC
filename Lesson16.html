Lesson 16: Images


  var img;
  var fontNice;

  function preload() {
     img = loadImage("christmastree.jpg");
     fontNice = loadFont("resources/PrincessSofia-Regular.ttf");

  }
  function setup() {
    createCanvas(windowWidth, windowHeight);
    background(255);
    img.resize(width, height);
    image(img, 0,50);
    textFont(fontNice);
    textAlign(CENTER);
    textSize(30);
    text("Merry Christmas", width/2,30)

  }
  function mousePressed(){
    image(img, mouseX, mouseY);
  }

  //decorate the christmastree with balls by your mouse

  var img;
  function preload(){
    img = loadImage("ria.png");
  }
  function setup() {
    createCanvas(windowWidth, windowHeight);
    background(100);
    img.resize(400,400);
    imageMode(CENTER);
    rectMode(CENTER);
  }

  function draw() {

    image(img,width/2,height/2);
    stroke(90, 0, 23);
    strokeWeight(10);
    fill(48, 70, 102)
    ellipse(width/2, 100,400,100);

    fill(140, 70, 106);
    rect(width/2,70,200,70);
    noStroke();
    ellipse(width/2, 100,190,50);
    strokeWeight(10);
    stroke(90, 0, 23);
    fill(48, 70, 102);
    ellipse(width/2, 35,200,50);

    strokeWeight(8);
    stroke(90, 0, 23)
    fill(48, 70, 102);
    ellipse((width/2) +  80, 80, 100,100);
    strokeWeight(4);
    stroke(140, 70, 106);
    fill(116, 4, 38);
    ellipse((width/2) +  110, 100, 100,100);
    ellipse((width/2) +  90, 110, 50,50);
    ellipse((width/2) +  80, 150, 30,30);
    ellipse((width/2) +  100, 170, 30,30);
  }
